# dotfiles

# Install

## create sym links
./install.sh

## vim bundle
* git submodule init
* git submodule update
* vim => :NeoBundleInstall

## add giftconfig
vim .gitconfig.local

> [user]  
>   name = "Your Name"  
>   email = you@example.com

# Uninstall

## remove sym links
./uninstall.sh
